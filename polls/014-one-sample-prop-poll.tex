\paragraph{Q1}

A fair coin is tossed 10 times. What is the probability that of getting exactly 5 heads?
\begin{itemize}
\item $\binom{10}{5} 0.5^5 (1-0.5)^5$ (\textbf{correct})
\item  $0.5^5$
\item  $5 \times 0.5^5 (1-0.5)^5$
\end{itemize}


\paragraph{Q2}
A fair coin is tossed 10 times. What is the probability that of getting less than 3 heads?
\begin{itemize}
	\item $\binom{10}{3} 0.5^3 (1-0.5)^7$ 
		\item $3 \times 0.5^3 (1-0.5)^7$
			\item $\sum_{k=0}^{3}\binom{10}{k} 0.5^k (1-0.5)^{10-k}$
	\item $\sum_{k=0}^{2}\binom{10}{k} 0.5^k (1-0.5)^{10-k}$ (\textbf{correct})
\end{itemize}

\paragraph{Q3}
A binomial distribution consists of 150 trials. If the probability of success on each trial is $0.4,$ what is the standard deviation of the distribution.
\begin{itemize}
	\item $\sqrt{150*0.4*0.6}$ \textbf{(correct)}
	\item $\sqrt{0.4*0.6 / 150}$
	\item $\sqrt{150*0.4*0.6}/\sqrt{150}$
\end{itemize}



\paragraph{Q4}
Which of the following is NOT required of a binomial distribution
\begin{itemize}
	\item Each trial has exactly two outcomes.
\item  There is a fixed number of trials.
\item The probability of success remains fixed for all trials.
\item There are more than 30 trials. (\textbf{correct})
\end{itemize}

\paragraph{Q5}
The mean $=n p$ and the standard deviation $=\sqrt{n p q}$ for
\begin{itemize}
	\item all probability distributions.
\item  normal distributions.
\item  binomial distributions. (\textbf{correct})
\item  none of the above
\end{itemize}
